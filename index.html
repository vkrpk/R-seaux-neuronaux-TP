import pandas as pd
import re
import unicodedata
from collections import Counter

def clean_user_review(text):
    if pd.isna(text):
        return ""

    # Convertir en string si ce n'est pas dj le cas
    text = str(text)

    try:
        # Normaliser Unicode
        text = unicodedata.normalize('NFKD', text).encode('ascii', 'ignore').decode('utf-8', 'ignore')
    except:
        text = ''.join(c for c in text if c.isprintable())

    # Rduire les rptitions excessives de lettres (ex : aaaaaa -> aa)
    text = re.sub(r'(.)\1{2,}', r'\1\1', text)

    # Mettre en minuscule
    text = text.lower()

    # Supprimer les liens URL
    text = re.sub(r'http[s]?://\S+', '', text)
    text = re.sub(r'www\.\S+', '', text)

    # Supprimer les motifs ASCII/moticnes
    ascii_patterns = [
        r'+', r'+', r'+', r'+',
        r'\(.*?\)', r'\(   \)',
        r'//.*?/', r'+', r'"+',
        r'REEEEE+', r'Ohgod(Ohgod)+',
        r'.*?', r'.*?',
    ]
    for pattern in ascii_patterns:
        text = re.sub(pattern, ' ', text)

    # Supprimer mentions inutiles
    text = re.sub(r'\b(early access review|spoiler alert|dev note)\b', ' ', text)

    # Supprimer ponctuation excessive (par exemple, "!!!", "....")
    text = re.sub(r'[\!\?]{2,}', ' ', text)

    # Supprimer les rptitions de groupes de mots
    text = re.sub(r'(\b[\w\s]{1,30}\b)(?:\s*\1){3,}', r'\1', text)

    # Supprimer les rptitions excessives de mots
    words = text.split()
    word_counts = Counter(words)
    filtered_words = [word for word in words if word_counts[word] < 5]
    text = ' '.join(filtered_words)

    # Supprimer les groupes de mots en majuscule de 2+ mots (ex : SHOUTING PHRASE)
    text = re.sub(r'\b(?:[A-Z]{2,}\s+){2,}[A-Z]{2,}\b', '', text)

    # Supprimer caractres non alphanumriques sauf ponctuation simple
    text = re.sub(r'[^\w\s.,!?]', ' ', text)

    # Nettoyer les parenthses et autres symboles inutiles
    text = re.sub(r'[()\"\']', ' ', text)

    # Rduire les espaces multiples
    text = re.sub(r'\s+', ' ', text).strip()

    return text

    # Nettoyer le texte
    cleaned_text = clean_user_review(text)
    # Convertir en squence
    sequence = tokenizer.texts_to_sequences([cleaned_text])
    # Padding
    padded = pad_sequences(sequence, maxlen=MAX_LENGTH, padding='post', truncating='post')
    # Prdiction
    prediction = model.predict(padded)[0][0]

    confidence = prediction if prediction > 0.5 else 1 - prediction
    return sentiment, confidence
